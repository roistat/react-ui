### Usage

```js
import Switcher from '@roistat/ui/lib/Switcher';
```
const View = require('../View').default;

*Switcher*

    <View>
        <Switcher />
    </View>

*Switcher is disabled*

    <View>
        <Switcher isDisabled={true} />
    </View>

*Switcher is checked*

    <View>
        <Switcher isChecked={true} />
    </View>
